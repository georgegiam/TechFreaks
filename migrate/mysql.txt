CREATE DATABASE IF NOT EXISTS diet;

use diet;

CREATE TABLE messages (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(30) NOT NULL,
email VARCHAR(30) NOT NULL,
subject VARCHAR(50),
message VARCHAR(255)
);

CREATE TABLE users (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(30) NOT NULL,
name VARCHAR(30) NOT NULL,
email VARCHAR(30),
password VARCHAR(255)
);
